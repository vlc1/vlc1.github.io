function rhs(t, q1, q2, q3) result(f)

	real(8), intent(in) :: t, q1, q2, q3
	real(8), dimension(3) :: f

	f(1) = q2
	f(2) = q3
	f(3) = q3 - 2. * q2 + q1 - 2.
